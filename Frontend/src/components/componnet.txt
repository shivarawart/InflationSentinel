{// src/components/BarterExchange.tsx
import { useState, useEffect } from 'react';
import api from '../lib/api';
import type { BarterListing } from '../types';
import { useAuth } from '../context/AuthContext';
import toast from 'react-hot-toast';

const BarterExchange = () => {
  const { user } = useAuth();
  const [listings, setListings] = useState<BarterListing[]>([]);
  const [myListings, setMyListings] = useState<BarterListing[]>([]);
  const [suggestedMatches, setSuggestedMatches] = useState<BarterListing[]>([]);
  const [activeTab, setActiveTab] = useState<'all' | 'my' | 'matches'>('all');
  const [showCreateModal, setShowCreateModal] = useState(false);
  const [loading, setLoading] = useState(true);

  const [newListing, setNewListing] = useState({
    type: 'offer' as 'offer' | 'request',
    title: '',
    description: '',
    tags: '',
  });

  // Load all listings
  useEffect(() => {
    fetchListings();
  }, []);

  const fetchListings = async () => {
    try {
      setLoading(true);
      const res = await api.get('/barter');
      setListings(res.data);

      // Filter my listings
      const mine = res.data.filter((l: BarterListing) => {
        const listingUserId = typeof l.user === 'object' && l.user !== null 
          ? (l.user as any)._id || (l.user as any).id 
          : l.user;
        return listingUserId === user?.id;
      });
      setMyListings(mine);
    } catch (err) {
      toast.error("Failed to load barter listings");
    } finally {
      setLoading(false);
    }
  };

  const createListing = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!newListing.title || !newListing.description) {
      toast.error("Title and description are required");
      return;
    }

    try {
      const tagsArray = newListing.tags
        ? newListing.tags.split(',').map(t => t.trim()).filter(Boolean)
        : [];

      await api.post('/barter', {
        type: newListing.type,
        title: newListing.title,
        description: newListing.description,
        tags: tagsArray,
        location: {
          city: user?.city || 'Tokyo',
          country: user?.country || 'JP'
        }
      });

      toast.success("Listing created successfully!");
      setShowCreateModal(false);
      setNewListing({ type: 'offer', title: '', description: '', tags: '' });
      fetchListings();
    } catch (err: any) {
      toast.error(err.response?.data?.message || "Failed to create listing");
    }
  };

  const findMatches = async (listingId: string) => {
    try {
      const res = await api.post(`/barter/${listingId}/match`);
      setSuggestedMatches(res.data);
      setActiveTab('matches');
      toast.success(`${res.data.length} potential matches found!`);
    } catch (err) {
      toast.error("No matches found at the moment");
    }
  };

  const filteredListings = activeTab === 'all' 
    ? listings 
    : activeTab === 'my' 
      ? myListings 
      : suggestedMatches;

  if (loading) {
    return (
      <div className="w-full min-h-screen flex items-center justify-center py-20 px-4 lg:px-8">
        <div className="text-slate-400 text-center">Loading barter marketplace...</div>
      </div>
    );
  }

  return (
    // 🔥 FULL WIDTH ROOT WRAPPER - This fixes desktop width issue
    <div className="w-full min-h-screen bg-gradient-to-br from-slate-900 via-slate-900/50 to-black px-4 sm:px-6 lg:px-8 xl:px-0 lg:-mx-4 xl:-mx-8">
      <div className="max-w-7xl mx-auto">
        {/* Header Section */}
        <div className="pt-12 pb-16 lg:pt-20 lg:pb-24">
          <div className="flex flex-col lg:flex-row justify-between items-start lg:items-end gap-8 lg:gap-0">
            <div className="flex-1">
              <h1 className="text-4xl md:text-5xl lg:text-6xl font-semibold tracking-tight bg-gradient-to-r from-white to-slate-200 bg-clip-text text-transparent">
                Barter Exchange
              </h1>
              <p className="text-slate-400 mt-4 text-lg md:text-xl max-w-2xl">
                Trade skills and goods. Beat inflation without cash.
              </p>
            </div>
            <button
              onClick={() => setShowCreateModal(true)}
              className="bg-emerald-600 hover:bg-emerald-700 px-8 py-4 md:py-4.5 rounded-3xl font-medium flex items-center gap-3 transition-all active:scale-95 shadow-2xl shadow-emerald-500/25 whitespace-nowrap"
            >
              <span className="text-xl">+</span> Create New Listing
            </button>
          </div>
        </div>

        {/* Tabs - Full Width */}
        <div className="border-b border-slate-700/50 backdrop-blur-sm bg-slate-900/30 rounded-2xl -mx-4 lg:-mx-8 px-6 lg:px-8 pb-2 sticky top-0 z-10 lg:top-4">
          <div className="flex overflow-x-auto lg:overflow-visible">
            {(['all', 'my', 'matches'] as const).map((tab) => (
              <button
                key={tab}
                onClick={() => setActiveTab(tab)}
                className={`flex-shrink-0 px-6 lg:px-10 py-5 font-medium text-lg transition-all border-b-4 rounded-xl -mb-px lg:-mb-2 whitespace-nowrap ${
                  activeTab === tab 
                    ? 'border-emerald-500 text-white bg-emerald-500/10 backdrop-blur-sm shadow-2xl shadow-emerald-500/25' 
                    : 'border-transparent text-slate-400 hover:text-slate-200 hover:bg-slate-800/50'
                }`}
              >
                {tab === 'all' && 'All Listings'}
                {tab === 'my' && 'My Listings'}
                {tab === 'matches' && 'Suggested Matches'}
              </button>
            ))}
          </div>
        </div>

        {/* Listings Grid - Full Width Responsive */}
        <div className="py-12 lg:py-20">
          {filteredListings.length === 0 ? (
            <div className="col-span-full text-center py-32 lg:py-48 text-slate-400">
              <div className="text-6xl mb-8">🤝</div>
              <h3 className="text-2xl lg:text-3xl font-semibold text-slate-200 mb-4">
                No listings found
              </h3>
              <p className="text-xl max-w-md mx-auto">
                {activeTab === 'all' && 'Be the first to create a barter listing!'}
                {activeTab === 'my' && 'Create your first listing to get started.'}
                {activeTab === 'matches' && 'No matches yet. Find matches for your listings.'}
              </p>
            </div>
          ) : (
            <div className="grid grid-cols-1 md:grid-cols-2 xl:grid-cols-3 2xl:grid-cols-4 gap-6 lg:gap-8">
              {filteredListings.map((listing) => (
                <div 
                  key={listing._id}
                  className="glass border border-slate-700/50 backdrop-blur-xl rounded-3xl p-7 hover:border-emerald-500/50 hover:shadow-2xl hover:shadow-emerald-500/20 transition-all group hover:-translate-y-2 h-full"
                >
                  <div className="flex items-center justify-between mb-6">
                    <span className={`px-5 py-2 text-xs font-semibold rounded-full uppercase tracking-wider shadow-lg ${
                      listing.type === 'offer' 
                        ? 'bg-gradient-to-r from-emerald-500/20 to-emerald-600/20 text-emerald-300 border border-emerald-500/30' 
                        : 'bg-gradient-to-r from-amber-500/20 to-amber-600/20 text-amber-300 border border-amber-500/30'
                    }`}>
                      {listing.type.toUpperCase()}
                    </span>
                    <span className="text-xs text-slate-500 font-medium bg-slate-800/50 px-3 py-1 rounded-xl">
                      {listing.location?.city}
                    </span>
                  </div>

                  <h3 className="text-xl lg:text-2xl font-semibold text-white mb-4 line-clamp-2 group-hover:text-emerald-400 transition-colors">
                    {listing.title}
                  </h3>
                  <p className="text-slate-400 text-[15px] line-clamp-3 mb-8 leading-relaxed">
                    {listing.description}
                  </p>

                  <div className="flex flex-wrap gap-2 mb-8">
                    {listing.tags?.slice(0, 4).map((tag, i) => (
                      <span key={i} className="text-xs bg-slate-800/50 backdrop-blur-sm px-4 py-2 rounded-2xl text-slate-300 border border-slate-700/50 hover:bg-slate-700/50 transition-all">
                        #{tag}
                      </span>
                    ))}
                    {listing.tags && listing.tags.length > 4 && (
                      <span className="text-xs bg-slate-800/50 px-4 py-2 rounded-2xl text-slate-500">
                        +{listing.tags.length - 4} more
                      </span>
                    )}
                  </div>

                  <button
                    onClick={() => findMatches(listing._id)}
                    className="w-full py-5 bg-gradient-to-r from-slate-800/50 to-slate-900/50 backdrop-blur-sm hover:from-emerald-600/90 hover:to-emerald-700/90 hover:border-emerald-500/50 border border-slate-700/50 rounded-3xl text-sm font-semibold transition-all group-hover:shadow-xl group-hover:shadow-emerald-500/25 text-white"
                  >
                    <span className="flex items-center justify-center gap-2">
                      Find Matches
                      <svg className="w-4 h-4 group-hover:translate-x-1 transition-transform" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M8 12h.01M12 12h.01M16 12h.01M21 12c0 4.418-4.03 8-9 8a9.863 9.863 0 01-4.255-.949L3 20l1.395-3.72C3.512 15.042 3 13.574 3 12c0-4.418 4.03-8 9-8s9 4.03 9 8z" />
                      </svg>
                    </span>
                  </button>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>

      {/* Create New Listing Modal - Full Screen Responsive */}
      {showCreateModal && (
        <div className="fixed inset-0 bg-black/80 backdrop-blur-md flex items-center justify-center z-50 p-4">
          <div className="bg-slate-900/95 backdrop-blur-2xl border border-slate-700/50 rounded-3xl w-full max-w-2xl mx-4 lg:mx-8 xl:max-w-4xl p-8 lg:p-12 shadow-2xl shadow-black/50 animate-in slide-in-from-bottom-4 duration-300">
            <div className="flex items-center justify-between mb-10">
              <h2 className="text-3xl lg:text-4xl font-bold bg-gradient-to-r from-white to-slate-200 bg-clip-text text-transparent">
                Create Barter Listing
              </h2>
              <button
                onClick={() => setShowCreateModal(false)}
                className="text-slate-400 hover:text-white p-2 rounded-2xl hover:bg-slate-800/50 transition-all"
              >
                <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
                </svg>
              </button>
            </div>
            
            <form onSubmit={createListing} className="space-y-8">
              <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
                <div>
                  <label className="block text-sm text-slate-400 font-medium mb-4">Type</label>
                  <select
                    value={newListing.type}
                    onChange={(e) => setNewListing({ ...newListing, type: e.target.value as 'offer' | 'request' })}
                    className="w-full bg-slate-800/50 backdrop-blur-sm border border-slate-700/50 rounded-3xl px-6 py-6 text-lg font-medium focus:border-emerald-500/50 focus:ring-4 focus:ring-emerald-500/20 transition-all"
                  >
                    <option value="offer">👨‍💼 I want to Offer / Give</option>
                    <option value="request">🙋‍♂️ I want to Request / Need</option>
                  </select>
                </div>

                <div>
                  <label className="block text-sm text-slate-400 font-medium mb-4">Location</label>
                  <div className="bg-slate-800/50 backdrop-blur-sm border border-slate-700/50 rounded-3xl px-6 py-6 text-lg font-medium text-slate-300">
                    {user?.city || 'Tokyo'}, {user?.country || 'JP'}
                  </div>
                </div>
              </div>

              <div>
                <label className="block text-sm text-slate-400 font-medium mb-4">Title</label>
                <input
                  type="text"
                  placeholder="e.g. English Conversation Lessons for Web Development Help"
                  value={newListing.title}
                  onChange={(e) => setNewListing({ ...newListing, title: e.target.value })}
                  className="w-full bg-slate-800/50 backdrop-blur-sm border border-slate-700/50 rounded-3xl px-6 py-6 text-xl font-semibold focus:border-emerald-500/50 focus:ring-4 focus:ring-emerald-500/20 transition-all"
                  required
                />
              </div>

              <div>
                <label className="block text-sm text-slate-400 font-medium mb-4">Description</label>
                <textarea
                  placeholder="Describe what you offer or need in detail. Be specific about what you're looking to trade for..."
                  value={newListing.description}
                  onChange={(e) => setNewListing({ ...newListing, description: e.target.value })}
                  className="w-full bg-slate-800/50 backdrop-blur-sm border border-slate-700/50 rounded-3xl px-6 py-6 text-lg resize-vertical min-h-[160px] focus:border-emerald-500/50 focus:ring-4 focus:ring-emerald-500/20 transition-all"
                  required
                />
              </div>

              <div>
                <label className="block text-sm text-slate-400 font-medium mb-4">Tags (comma separated)</label>
                <input
                  type="text"
                  placeholder="english, teaching, language, webdev, react, freelance"
                  value={newListing.tags}
                  onChange={(e) => setNewListing({ ...newListing, tags: e.target.value })}
                  className="w-full bg-slate-800/50 backdrop-blur-sm border border-slate-700/50 rounded-3xl px-6 py-6 text-lg focus:border-emerald-500/50 focus:ring-4 focus:ring-emerald-500/20 transition-all"
                />
              </div>

              <div className="flex flex-col sm:flex-row gap-4 pt-8">
                <button
                  type="button"
                  onClick={() => setShowCreateModal(false)}
                  className="flex-1 py-6 border border-slate-700/50 rounded-3xl font-semibold text-lg hover:bg-slate-800/50 backdrop-blur-sm transition-all hover:shadow-xl hover:shadow-slate-500/25"
                >
                  Cancel
                </button>
                <button
                  type="submit"
                  className="flex-1 py-6 bg-gradient-to-r from-emerald-600 to-emerald-700 hover:from-emerald-700 hover:to-emerald-800 rounded-3xl font-semibold text-lg shadow-2xl shadow-emerald-500/25 hover:shadow-emerald-500/40 transition-all active:scale-95"
                >
                  🚀 Publish Listing
                </button>
              </div>
            </form>
          </div>
        </div>
      )}
    </div>
  );
};

export default BarterExchange;
}

    setActiveItem(path);
  }, []);

  return (
    // 🔥 Premium Responsive Width Sidebar
    <div className="w-72 lg:w-80 xl:w-96 2xl:w-[28rem] flex-shrink-0 flex flex-col bg-gradient-to-b from-slate-950/90 via-slate-900/80 to-black/50 backdrop-blur-3xl border-r border-slate-800/50 shadow-2xl shadow-slate-900/50 group/sidebar hover:shadow-emerald-500/20 transition-all duration-500 overflow-hidden">
      
      {/* 🔥 Animated Logo Header */}
      <div className="p-8 lg:p-10 border-b border-slate-800/30 relative overflow-hidden">
        <div className="absolute inset-0 bg-gradient-to-r from-emerald-500/5 to-transparent -skew-x-12 -translate-x-full group-hover/sidebar:translate-x-full transition-transform duration-1000"></div>
        
        <div className="relative flex items-center gap-4 lg:gap-6">
          <div className="relative w-14 h-14 lg:w-16 lg:h-16 xl:w-20 xl:h-20 bg-gradient-to-br from-emerald-500 via-emerald-600 to-blue-600 rounded-3xl shadow-2xl shadow-emerald-500/40 flex items-center justify-center animate-spin-slow group-hover/sidebar:animate-none transition-all duration-700">
            <div className="w-8 h-8 lg:w-10 lg:h-10 xl:w-12 xl:h-12 bg-white/10 backdrop-blur-xl rounded-2xl flex items-center justify-center shadow-inner">
              <span className="text-2xl lg:text-3xl xl:text-4xl font-black text-white drop-shadow-lg">IS</span>
            </div>
          </div>
          
          <div className="min-w-0">
            <h1 className="text-2xl lg:text-3xl xl:text-4xl 2xl:text-5xl font-black tracking-tight bg-gradient-to-r from-white via-slate-100 to-emerald-200 bg-clip-text text-transparent drop-shadow-2xl leading-tight">
              InflationSentinel
            </h1>
            <p className="text-slate-400 text-sm lg:text-base xl:text-lg font-bold mt-1 lg:mt-2 tracking-wide uppercase">
              Beat the Rise
            </p>
          </div>
        </div>
      </div>

      {/* 🔥 Navigation - Full Width Responsive */}
      <nav className="flex-1 p-6 lg:p-8 xl:p-10 pt-0 lg:pt-2 overflow-y-auto scrollbar-thin scrollbar-thumb-slate-700/50 scrollbar-track-slate-900/30">
        <div className="space-y-2 lg:space-y-3">
          {navItems.map((item, index) => (
            <NavLink
              key={item.to}
              to={item.to}
              className={({ isActive }) =>

                  isActive

              }
              onClick={() => setActiveItem(item.to)}

              {/* Animated Icon Background */}
              <div className={`w-12 h-12 lg:w-14 lg:h-14 xl:w-16 xl:h-16 rounded-2xl flex items-center justify-center shadow-lg backdrop-blur-sm transition-all duration-300 ${
                activeItem === item.to 
                  ? `bg-gradient-to-r ${item.color} shadow-2xl shadow-[${item.color.replace('from-', '').replace('to-', '')}]/30 scale-110`
                  : 'bg-slate-800/50 group-hover/nav:bg-slate-700/70 border border-slate-700/50 group-hover/nav:border-slate-600/50'
              }`}>
                <span className="text-2xl lg:text-3xl xl:text-4xl drop-shadow-lg">{item.icon}</span>
              </div>
              
              {/* Label with Active Glow */}
              <div className="min-w-0 flex-1">
                <span className="text-lg lg:text-xl xl:text-2xl 2xl:text-3xl font-bold tracking-wide leading-tight block">
                  {item.label}
                </span>
                {activeItem === item.to && (
                  <div className="w-full h-1 bg-gradient-to-r from-emerald-400 to-emerald-500 rounded-full mt-2 lg:mt-3 animate-pulse shadow-lg shadow-emerald-500/50"></div>
                )}
              </div>
              
              {/* Active Badge */}
              {activeItem === item.to && (
                <div className="absolute -right-4 lg:-right-6 top-1/2 -translate-y-1/2 w-8 h-8 lg:w-10 lg:h-10 bg-emerald-500/20 backdrop-blur-xl rounded-full border-4 border-emerald-500/50 shadow-2xl shadow-emerald-500/30 animate-ping"></div>
              )}
            </NavLink>
          ))}
        </div>
      </nav>

      {/* 🔥 User Profile + Logout */}
      <div className="p-6 lg:p-8 xl:p-10 border-t border-slate-800/30 bg-slate-950/50 backdrop-blur-xl relative overflow-hidden">
        <div className="absolute inset-0 bg-gradient-to-b from-slate-800/30 to-transparent -skew-y-6 -rotate-1"></div>
        
        {user ? (
          <>
            <div className="relative bg-gradient-to-br from-slate-800/70 via-slate-800/40 to-slate-900/60 backdrop-blur-xl rounded-3xl p-6 lg:p-8 mb-6 lg:mb-8 border border-slate-700/50 shadow-xl shadow-slate-900/30 hover:shadow-emerald-500/20 transition-all duration-300 group/profile">

                  <span className="text-xl lg:text-2xl xl:text-3xl font-black text-white drop-shadow-lg">
                    {user.name?.charAt(0)?.toUpperCase() || 'U'}
                  </span>
                </div>
                <div className="min-w-0 flex-1">
                  <div className="text-lg lg:text-xl xl:text-2xl font-black text-white truncate mb-1 lg:mb-2">
                    {user.name || 'User'}
                  </div>
                  <div className="flex items-center gap-2 text-sm lg:text-base text-slate-400 font-medium">
                    <svg className="w-4 h-4 lg:w-5 lg:h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                      <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M17.657 16.657L13.414 20.9a1.998 1.998 0 01-2.827 0l-4.244-4.243a8 8 0 1111.314 0z" />
                      <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M15 11a3 3 0 11-6 0 3 3 0 016 0z" />
                    </svg>
                    <span className="truncate">{user.city}, {user.country}</span>
                  </div>
                </div>
              </div>
            </div>
            
            <button
              onClick={logout}
              className="group relative w-full py-5 lg:py-6 px-6 lg:px-8 text-lg lg:text-xl font-bold text-slate-400 hover:text-red-400 hover:bg-red-500/10 rounded-3xl border border-slate-700/50 hover:border-red-500/30 backdrop-blur-xl shadow-xl hover:shadow-red-500/20 transition-all duration-300 overflow-hidden"
            >
              <div className="absolute inset-0 bg-gradient-to-r from-red-500/10 to-transparent transform -skew-x-12 -translate-x-full group-hover:translate-x-full transition-transform duration-700"></div>
              <div className="relative flex items-center justify-center gap-3">
                <svg className="w-6 h-6 lg:w-7 lg:h-7" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M17 16l4-4m0 0l-4-4m4 4H7m6 4v1a3 3 0 01-3 3H6a3 3 0 01-3-3V7a3 3 0 013-3h4a3 3 0 013 3v1" />
                </svg>
                Sign Out
              </div>
            </button>
          </>
        ) : (
          <div className="text-center py-12 text-slate-500">
            <p className="text-lg font-medium mb-4">Join the fight against inflation</p>
            <button className="px-8 py-3 bg-emerald-600 hover:bg-emerald-700 rounded-2xl font-bold text-white transition-all">
              Sign In
            </button>
          </div>
        )}
      </div>
    </div>
  );
};

export default Sidebar;
